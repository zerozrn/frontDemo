js闭包

典型的闭包

    
        // 典型的闭包
        var scope="global scope";
        function f1() {
            var scope="local scope";
            function f2() {
                return scope;
            }
            return f2();
        }
        console.log(f1());

另外一种写法如下:

    
        // 对上述代码稍加改变，结果是。。。
        var scope1="global scope";
        function foo1() {
            var scope1="local scope";
            function foo2() {
                return scope1;
            }
            return foo2;
        }
        console.log(foo1()());

两种写法的结果均为local scope,可打开console查看结果
